<script src="https://cdn.jsdelivr.net/npm/canvas-confetti@1.5.1/dist/confetti.browser.min.js"></script>
<script>
    const music = document.getElementById('bgMusic');
    
    // 照片輪播數據
    const photoUrls = [
        "你的照片網址1",
        "你的照片網址2",
        "你的照片網址3"
    ];
    let photoIndex = 0;

    function startExperience() {
        music.play(); // 開始播放純音樂
        document.getElementById('step1').style.display = 'none';
        document.getElementById('step2').style.display = 'block';
        
        // 啟動照片輪播
        setInterval(() => {
            photoIndex = (photoIndex + 1) % photoUrls.length;
            document.getElementById('slideshow').src = photoUrls[photoIndex];
        }, 3000);
    }

    function showCake() {
        document.getElementById('step2').style.display = 'none';
        document.getElementById('step3').style.display = 'block';
    }

    function blowCandle() {
        document.getElementById('candleFlame').style.display = 'none';
        confetti({ particleCount: 200, spread: 80, origin: { y: 0.6 } });
        
        setTimeout(() => {
            document.getElementById('step3').style.display = 'none';
            document.getElementById('step4').style.display = 'block';
            startCountdown();
        }, 1500);
    }

    // 倒數計時邏輯 (7月9日)
    function startCountdown() {
        const targetDate = new Date("July 9, 2026 00:00:00").getTime();

        setInterval(() => {
            const now = new Date().getTime();
            const diff = targetDate - now;

            const days = Math.floor(diff / (1000 * 60 * 60 * 24));
            const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const mins = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
            const secs = Math.floor((diff % (1000 * 60)) / 1000);

            document.getElementById('timerDisplay').innerHTML = `
                <div class="timer-unit"><span class="timer-num">${days}</span>天</div>
                <div class="timer-unit"><span class="timer-num">${hours}</span>時</div>
                <div class="timer-unit"><span class="timer-num">${mins}</span>分</div>
                <div class="timer-unit"><span class="timer-num">${secs}</span>秒</div>
            `;
        }, 1000);
    }
</script>
